= Files

You can define a set of additional files that should be uploaded as part of the release. These files may also
be checksumed and signed before uploading.

include::partial$legend.adoc[]

[tabs]
====
YAML::
+
--
[source,yaml]
[subs="+macros"]
----
# icon:dot-circle[]
files:

  # Defines a list of artifacts.
  # Each entry must define a path: field.
  # The path may be relative to the project's basedir or absolute.
  # icon:dot-circle[] icon:file-alt[]
  artifacts:
    - path: path/to/some/file.txt
    - path: path/to/another/file.md

  # Defines a list of globs.
  # icon:dot-circle[]
  globs:
      # The pattern to use.
      # May use glob or regex pattern syntax.
      # icon:exclamation-triangle[] icon:file-alt[]
    - pattern: path/to/**/*.md

      # Additional properties used when evaluating templates.
      # icon:dot-circle[]
      extraProperties:
        # Key will be capitalized and prefixed with `artifact`, i.e, `artifactFoo`.
        foo: bar
----
--
TOML::
+
--
[source,toml]
[subs="+macros"]
----
# icon:dot-circle[]
[files]

  # Defines a list of artifacts.
  # Each entry must define a path: field.
  # The path may be relative to the project's basedir or absolute.
  # icon:dot-circle[] icon:file-alt[]
  artifact = [
    { path = "path/to/some/file.txt" },
    { path = "path/to/another/file.md" }
  ]

  # Defines a list of globs.
  # icon:dot-circle[]
  pass:[[[files.globs]]]
    # The pattern to use.
    # May use glob or regex pattern syntax.
    # icon:exclamation-triangle[] icon:file-alt[]
    pattern = "path/to/**/*.md"

    # Additional properties used when evaluating templates.
    # icon:dot-circle[]
    extraProperties.foo = "bar"
    # Key will be capitalized and prefixed with `artifact`, i.e, `artifactFoo`.
----

Maven::
+
--
[source,xml]
[subs="verbatim,+macros"]
----
<jreleaser>
  <!--
    icon:dot-circle[]
  -->
  <files>

    <!--
      Defines a list of files.
      Each entry must define a path: field.
      The path may be relative to the project's basedir or absolute.
      icon:dot-circle[] icon:file-alt[]
    -->
    <artifacts>
      <artifact>
        <path>path/to/some/file.txt</path>
      <artifact>
      </artifact>
        <path>path/to/another/file.md</path>
      </artifact>
    </artifacts>

    <!--
      Defines a list of globs.
      icon:dot-circle[]
    -->
    <globs>
      <glob>
        <!--
          The pattern to use.
          May use glob or regex pattern syntax.
          icon:exclamation-triangle[] icon:file-alt[]
        -->
        <pattern>path/to/**/*.md</pattern>

        <!--
          Additional properties used when evaluating templates.
          icon:dot-circle[] icon:file-alt[]
        -->
        <extraProperties>
          <!--
            Key will be capitalized and prefixed with `artifact`, i.e, `artifactFoo`.
          -->
          <foo>bar</foo>
        </extraProperties>
      </glob>
    </globs>
  </files>
</jreleaser>
----

Gradle::
+
--
[source,groovy]
[subs="+macros"]
----
jreleaser {
  // icon:dot-circle[]
  files {
    // Defines a file entry
    // The path may be relative to the project's basedir or absolute.
    // icon:dot-circle[] icon:file-alt[]
    artifact {
      path = 'path/to/some/file.txt'
    }
    artifact {
      path = 'path/to/another/file.md'
    }

    // Defines a glob entry
    // icon:dot-circle[]
    glob {
      // The pattern to use.
      // May use glob or regex pattern syntax.
      // icon:exclamation-triangle[] icon:file-alt[]
      template = 'path/to/**/*.md'

      // Additional properties used when evaluating templates.
      // Key will be capitalized and prefixed with `artifact`, i.e, `artifactFoo`.
      // icon:dot-circle[]
      extraProperties.put('foo', 'bar')
    }
  }
}
