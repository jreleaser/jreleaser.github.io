= Azure
:deployer_id:   azure
:deployer_name: Azure
:deployer_url:  pass:[https://pkgs.dev.azure.com/duke/_packaging/duke/maven/v1]
:default_auth:  BASIC
:deployer_sys_key: azure
:deployer_env_key: AZURE

Deploys staged artifacts to link:https://dev.azure.com[Azure DevOps].

.Prerequisites:

 * A user + password / token with grants to deploy packages.

IMPORTANT: This feature *does not* support deploying artifacts other than JARs and POMs to a Maven compatible repository.

WARNING: Publication of snapshots is *not* allowed.

== Example

[tabs]
====
YAML::
+
--
[source,yaml]
[subs="+macros,attributes"]
----
deploy:
  maven:
    {deployer_id}:
      app:
        active: ALWAYS
        url: {deployer_url}
        stagingRepositories:
          - target/staging-deploy
----
--
TOML::
+
--
[source,toml]
[subs="+macros,attributes"]
----
[deploy.maven.{deployer_id}.app]
  active = "ALWAYS"
  url = "{deployer_url}"
  stagingRepositories = ["target/staging-deploy"]
----
--
JSON::
+
--
[source,json]
[subs="+macros,attributes"]
----
{
  "jreleaser": {
    "deploy": {
      "maven": {
        "{deployer_id}": {
          "app": {
            "active": "ALWAYS",
            "url": "{deployer_url}",
            "stagingRepositories": [
              "target/staging-deploy"
            ]
          }
        }
      }  
    }
  }
}
----
--
Maven::
+
--
[source,xml]
[subs="+macros,attributes,verbatim"]
----
<jreleaser>
  <deploy>
    <maven>
      <{deployer_id}>
        <app>
          <active>ALWAYS</active>
          <url>{deployer_url}</url>
          <stagingRepositories>target/staging-deploy</stagingRepositories>
        </app>
      </{deployer_id}>
    </maven>
  </deploy>
</jreleaser>
----
--
Gradle::
+
--
[source,groovy]
[subs="+macros,attributes"]
----
jreleaser {
  deploy {
    maven {
      {deployer_id} {
        app {
          active = 'ALWAYS'
          url = '{deployer_url}'
          stagingRepository('target/staging-deploy')
        }
      }
    }
  }
}
----
